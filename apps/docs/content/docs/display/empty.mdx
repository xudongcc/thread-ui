## Installation

```bash
pnpm dlx shadcn@latest add @thread-ui/empty
```

## Usage

```tsx
import { Empty } from "@/components/thread-ui/empty";
```

```tsx
<Empty
  description="You haven't created any projects yet. Get started by creating your first project."
  icon={<FolderCodeIcon />}
  primaryAction={{ label: "Create Project" }}
  secondaryAction={{ label: "Import Project" }}
  title="No Projects Yet"
/>
```
